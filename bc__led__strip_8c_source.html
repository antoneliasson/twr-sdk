119 static void _bc_led_strip_task(void *param)
120 {
121  bc_led_strip_t *self = (bc_led_strip_t *)param;
122 
123  if (self->_effect.name == BC_LED_STRIP_EFFECT_NONE)
124  {
125  bc_scheduler_plan_current_absolute(BC_TICK_INFINITY);
126 
127  if (self->_event_handler != NULL)
128  {
129  self->_event_handler(self, BC_LED_STRIP_EVENT_EFFECT_DONE, self->_event_param);
130  }
131 
132  return;
133  }
134 
135  if (self->_effect.name == BC_LED_STRIP_EFFECT_TEST)
136  {
137  uint8_t intensity = 255 * (self->_effect.led + 1) / (self->_buffer->count + 1);
138 
139  if (self->_effect.round == 0)
140  {
141  self->_driver->set_pixel_rgbw(self->_effect.led, intensity, 0, 0, 0);
142  }
143  else if (self->_effect.round == 1)
144  {
145  self->_driver->set_pixel_rgbw(self->_effect.led, 0, intensity, 0, 0);
146  }
147  else if (self->_effect.round == 2)
148  {
149  self->_driver->set_pixel_rgbw(self->_effect.led, 0, 0, intensity, 0);
150  }
151  else if (self->_effect.round == 3)
152  {
153  if (self->_buffer->type == BC_LED_STRIP_TYPE_RGBW)
154  {
155  self->_driver->set_pixel_rgbw(self->_effect.led, 0, 0, 0, intensity);
156  }
157  else
158  {
159  self->_driver->set_pixel_rgbw(self->_effect.led, intensity, intensity, intensity, 0);
160  }
161  }
162  else
163  {
164  self->_driver->set_pixel_rgbw(self->_effect.led, 0, 0, 0, 0);
165  }
166 
167  self->_effect.led++;
168 
169  if (self->_effect.led == self->_buffer->count)
170  {
171  self->_effect.led = 0;
172 
173  self->_effect.round++;
174 
175  if (self->_effect.round == 5)
176  {
177  self->_effect.name = BC_LED_STRIP_EFFECT_NONE;
178  }
179  }
180  }
181  else if (self->_effect.name == BC_LED_STRIP_EFFECT_RAINBOW_CYCLE)
182  {

189  }
