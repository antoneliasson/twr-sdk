167 static void ws2812b_send()
168 {
169  // transmission complete flag
170  ws2812b.transfer_complete = 0;
171 
172  uint32_t i;
173  uint32_t j;
174 
175  for (i = 0; i < WS2812_BUFFER_COUNT; i++)
176  {
177  ws2812b.item[i].frame_buffer_counter = 0;
178  ws2812b.item[i].frame_buffer_size = bc_module_power.led_strip_mode * bc_module_power.led_strip_count;
179 
180  for (j = 0; j < BIT_BUFFER_LEDS; j++)
181  {
182  load_next_framebuffer_data(&ws2812b.item[i], j); // ROW j
183  }
184  }
185 
186  HAL_TIM_Base_Stop(&timer2_handle);
187  (&timer2_handle)->Instance->CR1 &= ~((0x1U << (0U)));
188 
189  // clear all DMA flags
190  __HAL_DMA_CLEAR_FLAG(&dmaUpdate, DMA_FLAG_TC2 | DMA_FLAG_HT2 | DMA_FLAG_TE2);
191 
192  // configure the number of bytes to be transferred by the DMA controller
193  dmaUpdate.Instance->CNDTR = 8 * bc_module_power.led_strip_mode * BIT_BUFFER_LEDS;
194 
195  // clear all TIM2 flags
196  __HAL_TIM_CLEAR_FLAG(&timer2_handle, TIM_FLAG_UPDATE | TIM_FLAG_CC1 | TIM_FLAG_CC2 | TIM_FLAG_CC3 | TIM_FLAG_CC4);
197 
198  // enable DMA channels
199  __HAL_DMA_ENABLE(&dmaUpdate);
200 
201  // IMPORTANT: enable the TIM2 DMA requests AFTER enabling the DMA channels!
202  __HAL_TIM_ENABLE_DMA(&timer2_handle, TIM_DMA_UPDATE);
203 
204  TIM2->CNT = timer_period - 1;
205 
206  // Set zero length for first pulse because the first bit loads after first TIM_UP
207  TIM2->CCR2 = 0;
208 
209  // Enable PWM Compare 1
210  //(&timer2_handle)->Instance->CCMR1 |= TIM_CCMR1_OC1M_1;
211  // Enable PWM Compare 2
212  (&timer2_handle)->Instance->CCMR1 |= TIM_CCMR1_OC2M_1;
213 
214  __HAL_DBGMCU_FREEZE_TIM2();
215 
216  // start TIM2
217  __HAL_TIM_ENABLE(&timer2_handle);
218 }

221 {
222  (void)dma_handle;
223 
224  // Is this the last LED?
225  if (ws2812b.repeat_counter != (bc_module_power.led_strip_count / 2 - 1))
226  {

248 void dma_transfer_complete_handler(DMA_HandleTypeDef *dma_handle)
249 {
250  (void)dma_handle;
251 
252  ws2812b.repeat_counter++;
253 
254  if (ws2812b.repeat_counter == bc_module_power.led_strip_count / 2)
255  {
256  // Transfer of all LEDs is done, disable DMA but enable tiemr update IRQ to stop the 50us pulse
257  ws2812b.repeat_counter = 0;
258 
259 
260 
261  // Stop timer
262  TIM2->CR1 &= ~TIM_CR1_CEN;
263 
264  // Disable DMA
265  __HAL_DMA_DISABLE(&dmaUpdate);
266  // Disable the DMA requests
267  __HAL_TIM_DISABLE_DMA(&timer2_handle, TIM_DMA_UPDATE);
268 
269  // Disable PWM output compare 1
270  //(&timer2_handle)->Instance->CCMR1 &= ~(TIM_CCMR1_OC1M_Msk);
271  //(&timer2_handle)->Instance->CCMR1 |= TIM_CCMR1_OC1M_2;
272 
273  // Disable PWM output Compare 2
274  (&timer2_handle)->Instance->CCMR1 &= ~(TIM_CCMR1_OC2M_Msk);
275  (&timer2_handle)->Instance->CCMR1 |= TIM_CCMR1_OC2M_2;
276 
277  // Set 50us period for Treset pulse
278  //TIM2->PSC = 1000; // For this long period we need prescaler 1000
279  TIM2->ARR = timer_reset_pulse_period;
280  // Reset the timer
281  TIM2->CNT = 0;
282 
283  // Generate an update event to reload the prescaler value immediately
284  TIM2->EGR = TIM_EGR_UG;
285  __HAL_TIM_CLEAR_FLAG(&timer2_handle, TIM_FLAG_UPDATE);
286 
287  // Enable TIM2 Update interrupt for Treset signal
288  __HAL_TIM_ENABLE_IT(&timer2_handle, TIM_IT_UPDATE);
289  // Enable timer
290  TIM2->CR1 |= TIM_CR1_CEN;
291 
292  }
293  else
294  {
295 
296  // Load bitbuffer with next RGB LED values
297  uint32_t i;
298  uint32_t j;
299  for (i = 0; i < WS2812_BUFFER_COUNT; i++)
300  {
301  for (j = (BIT_BUFFER_LEDS / 2); j < BIT_BUFFER_LEDS; j++)
302  {
303  load_next_framebuffer_data(&ws2812b.item[i], j);
304  }
305  }
306 
307  }
